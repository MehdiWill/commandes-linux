alias : 
    win : go to win folder

cat: voir le contenu d'un fichier


ls : voir le contenu d'un dossier
   -a : voir fichier caché
   -l : afficher sous forme de liste 
   -R : affiche les enfants

cd : se deplacer
    / : dossier racine
    ~ : mes dossiers
    ..: dossier parent

mv : deplacer un dossier/fichier

cp : copier /coller
   -r : pour dossier

mkdir : creer un dossier
touch : creer un fichier

rm : supprimer
    -f : force
    -r : supprimer dossier

pwd : afficher le chemin du dossier

sudo: paser en admin
apt install : installer un package (necessite sudo)

clear : clean la console

npm :
   install
      -g : globale
   uninstall
      -g


GIT
    status
    clone : dl un repo
    log :(q pour quitter)
    diff: (q pour quitter)
    
    add:
        . : tout le dossier
        {nom}: nom du fichier/dossier 
    commit

    pull: mise a jour avant transformation et/ou push
    push: envoyer le dossier vers github
    log : afficher les commits (q pour quitter)
    stash: mettre de coté
    stash pop: replacer
